    agent any 
    stages {
        stage('clean') { 
            steps {
                sh 'mvn clean'
            }
        }
        stage('validate') { 
            steps {
                sh 'mvn validate' 
            }
        }
        stage('Test') { 
            steps {
                sh 'mvn test' 
            }
        }
        stage('Build') { 
            steps {
                sh 'mvn package' 
            }
        }
    }
}
